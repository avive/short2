Shirley Shor art Tokens (ShorTs) are issued and maintained on the Ethereum public ledger. Purchasers of Team Human art pieces will receive ShorTs based on their purchase amount. For example, one ShorT for every 1 Eth spent on purchasing art from the artist. The artist herby promises to sell her art work exclusively to token purchases for a period of the next 5 years. Only a fixed amount of 3500 ShortS exist and no new ones may be created. The artist may increase the price of each token in the future but will and can not create new ones. ShortTs holders become part of Team Human and get exclusive benefits and rights:

  • A right to receive a Team Human Certificate and a share of the profits from the amount of editions sold
  • First right to purchase a specific exclusive limited edition of the Team Human artwork. This includes the video real-time-generated piece, prints and other related material that will be emerging organically from this project
  • Right to participate in future Team Human art pieces
  • Participate in the artistic creative process

Smart Contract

    pragma solidity >=0.5.0;
    import "openzeppelin-solidity/contracts/token/ERC20/ERC20.sol";
    import "openzeppelin-solidity/contracts/token/ERC20/ERC20Detailed.sol";
    import "openzeppelin-solidity/contracts/math/SafeMath.sol";
    import "openzeppelin-solidity/contracts/ownership/Ownable.sol";
    import "openzeppelin-solidity/contracts/payment/PullPayment.sol";

    contract ShirleyShorToken is ERC20, ERC20Detailed, Ownable, PullPayment {
        uint public tokenPrice = 1 ether;
        constructor() ERC20Detailed("Shirley Shor Token", "ShorT", 0) public {
            _mint(msg.sender, 3500);
        }
        function purchase(uint tokens) public payable {
            require(msg.value == tokens.mul(tokenPrice));
            _transfer(owner(), msg.sender, tokens);
            _asyncTransfer(owner(), msg.value);
        }
        function updatePrice(uint price) onlyOwner public {
            require(price > tokenPrice);
            tokenPrice = price;
        }
    }
